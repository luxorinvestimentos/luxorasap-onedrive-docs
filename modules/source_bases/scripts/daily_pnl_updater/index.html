    def create_daily_pnl_db(lq: LuxorQuery, portfolios: dict, today: dt.date,
                         bdays: int = 30)->pd.DataFrame:
    """Cria base diária de PnL para diversos portfolios (definidos na funcao).

    Args:
        lq (LuxorQuery): objeto LuxorQuery para realizar queries no banco de dados.
        portfolios (dict): dicionario com os portfolios e sua configuracao.
        today (dt.date): data fim, referencia para calculo da data inicial
        bdays (int, optional): numero de dias uteis anteriores a ref_date.
                            Será usado para achar a data inicial. Defaults to 30.

    Returns:
        pd.DataFrame: Base diária de PnL.
    """
    daily_pnls = []
    # Criando base de pnl diario
    for df_name, params in portfolios.items():
        for currency in ['brl', 'usd']:
            df = lq.calculate_daily_pnl(
                params['fund_ref'], ref_date=today,
                classification_filters=params['classification_filter'],
                currency=currency, bdays=bdays, holiday_location="all", asset_filters=params['asset_filter'],
                group_filters=params['group_filter'], location_filters=params['location_filter'],
                include_cash=params['include_cash'], include_cash_debt=params['include_cash_debt'],
                currency_exposure_filters=params['currency_exposure'],
                annual_adm_fee=params['fees'], usdbrl_ticker="usdbrl cmpl curncy").rename(columns={"Today":"Date"})
            df["Currency"] = currency
            df["Series_Alias"] = df_name

            #daily_fee = (1 + params['fees'])**(1/365) - 1
            daily_pnls.append(df.copy())

    daily_pnls = pd.concat(daily_pnls)

    return daily_pnls
