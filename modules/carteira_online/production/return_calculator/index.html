    return_calculator.py¶

    Calcula a cota gerencial diária, além dos retornos 1D, MTD e YTD.

    Script de atualizacao da base historica de retornos.

    • A cada mes e ano devemos alterar o preco da cota em 'self.last_quotas'
    • As alteracoes de # cotas onshore eh lida automaticamente das carteiras do BTG
    • As alteracoes de # cotas offshore precisam ser alteradas na mão (em hist_returns.xlsx)

    Fluxo: 1. Carragamento das bases de hoje e do dia anterior e carregamento da base historica 2. Base historica lida é segregada por fundo em um dicionario 3. Atualização dos fundos onshore: 3.1 Leitura das carteiras do BTG mais atual para resgatar qtd de cotas 3.2 Insere linha nova com dados do btg ou altera a #_cotas, recalculando cota e propagando alteracao 4. 5. 6.

    def get_prev_pft_path(self, directory):
        """Buscamos no diretorio a base de portfolios mais recente disponivel (sem ser o de hoje)"""

        # Extraindo data do nome de cada arquivo e encontrando a maior menor do que hoje
        files = os.listdir(directory)
        #today = dt.datetime.today()
        today = dt.datetime(self.today.year, self.today.month, self.today.day)
        dates = [ dt.datetime.strptime(file.split("_")[-1].replace(".xlsx",""), "%Y-%m-%d") for file in files]
        dates = list(filter(lambda base_date : base_date < today, dates))
        max_date = max(dates)

        if today.weekday() != 0:
            if (today - max_date).days > 1:
                logger.warning("ATENÇÃO: Base de portfolios anterior esta desatualizada!")
        else:
            if (today - max_date).days > 3:
                logger.warning("ATENÇÃO: Base de portfolios anterior esta desatualizada!")

        file = "base_portfolios_" + max_date.strftime("%Y-%m-%d") + ".xlsx"
        return os.path.join(directory, file)

    def calculate_quota(self):

        funds_cur_quota_data = {}

        for fund in self.fund_names:
            if fund == "lipizzaner usd" or fund == "mangalarga fic fia usd" or fund == "maratona usd" or fund == "fund b brl": continue
            hist_data = self.query_mngr.get_table(fund.replace(" ","_")+"_quotas")

            prev_data_entry = hist_data.loc[hist_data["Date"].dt.date <= self.today].tail(1)
            prev_quota_amnt = prev_data_entry["#"].squeeze()

            aum_adjust = self.fund_pl_variation_adjust[fund]

            if aum_adjust != 0:
                logger.warning(f"ATENÇÃO: PL do {fund} ajustado em {aum_adjust}.")
                #print(f"ATENÇÃO: PL do {fund} ajustado em {aum_adjust}.")

            cur_aum = self.calculate_pls(fund) + aum_adjust
            cur_quota_value = cur_aum/prev_quota_amnt

            funds_cur_quota_data[fund] = {"amount" : prev_quota_amnt, "quota" : cur_quota_value}
            #print(fund,  funds_cur_quota_data)

        return funds_cur_quota_data


    def calculate_pls(self, fund_name):

        # correcao para mensal e anual
        base = self.ptf_bases[fund_name.replace(" ", "_")]



        # Para o manga fic fia, vamos pegar o pl em tempo real do manga master e somar com o caixa presente no fic fia
        if fund_name == "mangalarga fic fia":
            lipizzaner_pl = (self.calculate_pls("lipizzaner") + self.fund_pl_variation_adjust["lipizzaner"] ) * self.query_mngr.calculate_fund_particip("mangalarga fic fia", "lipizzaner", self.today)
            #print(f"manga fic: {base.loc[base["Asset"] != "lipizzaner", "Mkt_Value"].sum()}")
            return lipizzaner_pl + base.loc[base["Asset"] != "lipizzaner", "Mkt_Value"].sum()

        return base["Mkt_Value"].sum()

                ret_d = (1+ret_d)/(1+usd_1d)-1

                fund = fund + " usd"

        # Calcular atribuicao de performance para cada fundo
        for fund_name in self.fund_names:

                is_manga_fic = True
                fund_name = "lipizzaner"

            f_key = fund_name.replace(" ", "_")
            base = self.prev_ptf_bases[f_key]

            # Formatando dataframes com colunas que vamos precisar
            perf_attr_base = base[ ["Nome", "Fundo", "Asset", "Ticker", "Quantidade", "Ticker_BBG", "Tipo", "Grupo", "Location", "%_pl"]].copy(deep=False)

            perf_attr_base["Fundo"] = fund_name

            usdbrl_pct_chg = self.query_mngr.get_pct_change("usdbrl curncy", recent_date=self.today, previous_date=self.yesterday)#self.query_mngr.get_bday_offset(self.today, -1))

            def __get_pct_var(row):
                '''Obtem 1 percentual de variacao diaria baseado no tipo de ativo.'''

                # JAMAIS USAR row["Fundo"] AQUI DENTRO
                if row["Grupo"] in ["caixa_us", "vc", "loan", "bond", "vc agro"]:
                    #TODO: consigo adicionar o vc no attribution diario, queremos fazer isso?
                    if fund_name in self.offshore:
                        return 0
                    else:
                        return usdbrl_pct_chg

                # Economizamos processamento, nao precisamos calcular para ativos que nao estamos posicionados
                if row["Quantidade"] == 0:
                    return 0

                # Tentamos primeiro obter variacao usando ticker bbg
                pct_change = self.query_mngr.get_pct_change(row["Ticker_BBG"], recent_date=self.today, previous_date=self.yesterday)#self.query_mngr.get_bday_offset(self.today, -1))

                #    pct_change = (1 + pct_change)/(1 + usdbrl_pct_chg)-1 # Converte para usd

                # Se nao conseguimos obter a variacao, sera setada para 0
                if pct_change is None:
                    return 0

                # para fundos onshore com ativos offshore, convertemos a variacao obtida pelo dolar
                if (row["Location"] == "us" and fund_name in self.onshore) or (row["Tipo"] == "ações_bdr"):
                    if row["Asset"] != "spx hawker": #Hawker ja esta com a cota corrigida pelo dolar!

                if row["Tipo"] != "swap_exp":

        usdbrl_pct_chg = self.query_mngr.get_pct_change("usdbrl curncy", recent_date=self.today, previous_date=self.yesterday)#self.query_mngr.get_bday_offset(self.today, -1))

        def __get_pct_var(row):
            '''Obtem 1 percentual de variacao diaria baseado no tipo de ativo.'''

            # JAMAIS USAR row["Fundo"] AQUI DENTRO
            if row["Grupo"] in ["caixa_us", "vc", "loan", "bond", "vc agro"]:
                #TODO: consigo adicionar o vc no attribution diario, queremos fazer isso?
                if fund_name in self.offshore:
                    return 0
                else:
                    return usdbrl_pct_chg

            # Economizamos processamento, nao precisamos calcular para ativos que nao estamos posicionados
            if row["Quantidade"] == 0:
                return 0

            # Tentamos primeiro obter variacao usando ticker bbg
            pct_change = self.query_mngr.get_pct_change(row["Ticker_BBG"], recent_date=self.today, previous_date=self.yesterday)#self.query_mngr.get_bday_offset(self.today, -1))
            #if row["Ticker_BBG"] == 'lzrfy us equity': #tratando retorno da adr de localiza
            #    pct_change = self.query_mngr.get_pct_change('rent3 bz equity', recent_date=self.today, previous_date=self.yesterday)
            #    pct_change = (1 + pct_change)/(1 + usdbrl_pct_chg)-1 # Converte para usd
            # Se nao for possivel, entao tentamos obter usando a coluna ticker
            if pct_change == 0:
                pct_change = self.query_mngr.get_pct_change(row["Ticker"], recent_date=self.today, previous_date=self.yesterday)#self.query_mngr.get_bday_offset(self.today, -1))
            # Caso especifico onde o diario se da pela variacao de omega
            if row["Ticker"] == "poraque fic fip":
                pct_change = self.query_mngr.get_pct_change("mega3 bz equity", recent_date=self.today, previous_date=self.yesterday)#self.query_mngr.get_bday_offset(self.today, -1))

            # Se nao conseguimos obter a variacao, sera setada para 0
            if pct_change is None:
                return 0

            # para fundos onshore com ativos offshore, convertemos a variacao obtida pelo dolar
            if (row["Location"] == "us" and fund_name in self.onshore) or (row["Tipo"] == "ações_bdr"):
                if row["Asset"] != "spx hawker": #Hawker ja esta com a cota corrigida pelo dolar!
                    # corrigir pelo dolar
                    pct_change = pct_change + usdbrl_pct_chg + pct_change * usdbrl_pct_chg

            return pct_change

        def __calculate_perf_attr(row):
            if row["Tipo"] != "swap_exp":
                return row["%_var"] * row["%_pl"]
            else:
                #sobe qndo o dolar cai e cai qndo o dolar sobe
                return -row["Quantidade"] * usdbrl_pct_chg/self.calculate_pls(fund_name)

        # Obter variacao percentual por ativo
        perf_attr_base["%_var"] = perf_attr_base.apply(lambda row:  __get_pct_var(row), axis=1)
        # Calcular attribution diario
        perf_attr_base["Perf_attr"] = perf_attr_base.apply(lambda row: __calculate_perf_attr(row), axis=1)
        # Remover colunas desnecessarias
        perf_attr_base = perf_attr_base.drop("Quantidade", axis=1)
        # Salvando resultado
        if is_manga_fic: # Foi calculado a partir do Lipizzaner
            # Precisamos ajustar pelo caixa presente no fundo
            perf_attr_base["Fundo"] = "mangalarga fic fia"
            # obtendo portfolio do manga
            manga_ptf = self.ptf_bases["mangalarga_fic_fia"]
            # obtendo quando do portfolio eh caixa
            cash_proportion = (manga_ptf.loc[manga_ptf["Asset"] != "lipizzaner", "Mkt_Value"].sum()) / self.calculate_pls("mangalarga fic fia")
            # Vamos ajustar o attribution do manga pelo inverso da proporcao de caixa
            perf_attr_base["Perf_attr"] = perf_attr_base["Perf_attr"].apply(lambda x: x * (1-cash_proportion))

            self.perf_attr_bases["mangalarga fic fia"] = perf_attr_base
        else:
            self.perf_attr_bases[fund_name] = perf_attr_base

        location = "bz" if fund in self.onshore else "us"

            usd_mtd = self.query_mngr.get_pct_change("usdbrl curncy", recent_date=self.today, previous_date=self.last_month_end)

    def get_prev_pft_path(self, directory):
    """Buscamos no diretorio a base de portfolios mais recente disponivel (sem ser o de hoje)"""

    # Extraindo data do nome de cada arquivo e encontrando a maior menor do que hoje
    files = os.listdir(directory)
    #today = dt.datetime.today()
    today = dt.datetime(self.today.year, self.today.month, self.today.day)
    dates = [ dt.datetime.strptime(file.split("_")[-1].replace(".xlsx",""), "%Y-%m-%d") for file in files]
    dates = list(filter(lambda base_date : base_date < today, dates))
    max_date = max(dates)

    if today.weekday() != 0:
        if (today - max_date).days > 1:
            logger.warning("ATENÇÃO: Base de portfolios anterior esta desatualizada!")
    else:
        if (today - max_date).days > 3:
            logger.warning("ATENÇÃO: Base de portfolios anterior esta desatualizada!")

    file = "base_portfolios_" + max_date.strftime("%Y-%m-%d") + ".xlsx"
    return os.path.join(directory, file)
